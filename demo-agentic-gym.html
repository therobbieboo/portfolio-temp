class AgenticGym: def __init__(self, model, config): self.model = model self.evaluator = RewardModel() self.buffer = ExperienceBuffer() def train_step(self, tasks): # 1. Generate agent trajectories trajectories = self.model.generate(tasks) # 2. Evaluate with reward model rewards = self.evaluator.score(trajectories) # 3. Update with PPO/DPO self.model.update(trajectories, rewards) return rewards.mean()
